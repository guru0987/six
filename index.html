Banker's Algorithm (C Program)


#include <stdio.h>

int num_processes, num_resources;

// Function to check if the system is in a safe state
int is_safe(int available[], int max[][10], int allocation[][10], int need[][10], int safe_sequence[]) {
    int work[num_resources];
    int finish[num_processes];

    // Initialize work with available resources
    for (int i = 0; i < num_resources; i++) {
        work[i] = available[i];
    }
    
    // Initialize finish array
    for (int i = 0; i < num_processes; i++) {
        finish[i] = 0;
    }

    int count = 0;
    while (count < num_processes) {
        int found = 0;
        for (int i = 0; i < num_processes; i++) {
            if (finish[i] == 0) {
                int j;
                for (j = 0; j < num_resources; j++) {
                    if (need[i][j] > work[j]) {
                        break;
                    }
                }
                
                // If process i can be run
                if (j == num_resources) {
                    for (int k = 0; k < num_resources; k++) {
                        work[k] += allocation[i][k];
                    }
                    safe_sequence[count++] = i;
                    finish[i] = 1;
                    found = 1;
                }
            }
        }
        
        // No safe sequence found
        if (found == 0) {
            return 0; 
        }
    }
    
    return 1; // System is in a safe state
}

int main() {
    printf("Enter the number of processes: ");
    scanf("%d", &num_processes);
    printf("Enter the number of resources: ");
    scanf("%d", &num_resources);

    int available[num_resources];
    int max[num_processes][10];
    int allocation[num_processes][10];
    int need[num_processes][10];
    int safe_sequence[num_processes];

    printf("\nEnter available resources:\n");
    for (int i = 0; i < num_resources; i++) {
        printf("Resource %d: ", i);
        scanf("%d", &available[i]);
    }

    printf("\nEnter Allocation Matrix:\n");
    for (int i = 0; i < num_processes; i++) {
        printf("Process %d: ", i);
        for (int j = 0; j < num_resources; j++) {
            scanf("%d", &allocation[i][j]);
        }
    }

    printf("\nEnter Max Matrix:\n");
    for (int i = 0; i < num_processes; i++) {
        printf("Process %d: ", i);
        for (int j = 0; j < num_resources; j++) {
            scanf("%d", &max[i][j]);
            need[i][j] = max[i][j] - allocation[i][j];
        }
    }
    
    // Check if the system is in a safe state
    if (is_safe(available, max, allocation, need, safe_sequence)) {
        printf("\nSystem is in a SAFE state. Safe sequence: ");
        for (int i = 0; i < num_processes; i++) {
            printf("P%d", safe_sequence[i]);
            if (i < num_processes - 1) {
                printf(" -> ");
            }
        }
        printf("\n");
    } else {
        printf("\nSystem is in an UNSAFE state. Deadlock is possible.\n");
    }

    return 0;
}
